se_keyboard_toggle_shift = {
	if = {
		limit = { has_country_flag = se_shift_enabled }
		remove_country_flag = se_shift_enabled
	}
	else = { set_country_flag = se_shift_enabled }
}
se_keyboard_disable_shift = {
	if = {
		limit = { has_country_flag = se_shift_enabled }
		remove_country_flag = se_shift_enabled
	}
}
se_keyboard_toggle_caps = {
	if = {
		limit = { has_country_flag = se_caps_enabled }
		remove_country_flag = se_caps_enabled
	}
	else = { set_country_flag = se_caps_enabled }
}
