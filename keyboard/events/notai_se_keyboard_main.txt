namespace = se_keyboard

country_event = {
	id = se_keyboard.0
	is_triggered_only = yes
	hide_window = no
	title = se_keyboard_title
	desc = se_keyboard_desc
	trigger = {
		is_ai = no
		se_is_keyboard_active_trigger = yes
	}
	# Setting up necessary values and ensuring functionality for the keyboard
	immediate = {
		set_country_flag = se_keyboard_enabled
		se_get_or_create_database = { bound_to = this }
		event_target:se_return_value = { save_event_target_as = se_player_database }
		se_clear_return_target = yes
		se_new_comparable_string = {
			identifier = se_player_input
			overwrite = yes
		}
		event_target:se_return_value = { save_event_target_as = se_keyboard_desc_string }
		se_get_list_header = { identifier = se_player_input }
		event_target:se_return_value = { save_event_target_as = se_keyboard_input_metadata }
		event_target:se_player_database = {
			se_create_database_entry = {
				identifier = se_player_input_final
				overwrite = no
			}
			event_target:se_return_value = {
				set_name = ""
				save_event_target_as = se_keyboard_input
			}
		}
		set_variable = {
			which = se_keyboard_current_index
			value = 0
		}
	}

				limit = { se_keyboard_is_capitals_enabled_trigger = yes }

			se_increment = { variable = se_keyboard_current_index }
