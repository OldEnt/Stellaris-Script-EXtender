se_keyboard_object_locked = {
	hidden_trigger = {
		if = {
			limit = { is_scope_type = megastructure }
			has_megastructure_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = planet }
			has_planet_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = planet }
			has_planet_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = country }
			has_country_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = ship }
			has_ship_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = fleet }
			has_fleet_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = galactic_object }
			has_star_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = leader }
			has_leader_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = pop_faction }
			has_pop_faction_flag = se_keyboard_locked
		}
		else_if = {
			limit = { is_scope_type = federation }
			has_federation_flag = se_keyboard_locked
		}
	}
}
