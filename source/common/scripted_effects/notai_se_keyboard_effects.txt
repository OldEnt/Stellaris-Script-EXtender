### Locks the current object as being written by the keyboard
se_keyboard_lock_object = {
	switch = {
		trigger = is_scope_type
		megastructure = { set_megastructure_flag = se_keyboard_locked }
		planet = { set_planet_flag = se_keyboard_locked }
		country = { set_country_flag = se_keyboard_locked }
		ship = { set_ship_flag = se_keyboard_locked }
		fleet = { set_fleet_flag = se_keyboard_locked }
		galactic_object = { set_star_flag = se_keyboard_locked }
		leader = { set_leader_flag = se_keyboard_locked }
		pop_faction = { set_pop_faction_flag = se_keyboard_locked }
		federation = { set_federation_flag = se_keyboard_locked }
		default = { log_error = "se_keyboard_lock_object: Cannot properly lock this scope type" }
	}
}
se_keyboard_unlock_object = {
	switch = {
		trigger = is_scope_type
		megastructure = { remove_megastructure_flag = se_keyboard_locked }
		planet = { remove_planet_flag = se_keyboard_locked }
		country = { remove_country_flag = se_keyboard_locked }
		ship = { remove_ship_flag = se_keyboard_locked }
		fleet = { remove_fleet_flag = se_keyboard_locked }
		galactic_object = { remove_star_flag = se_keyboard_locked }
		leader = { remove_leader_flag = se_keyboard_locked }
		pop_faction = { remove_pop_faction_flag = se_keyboard_locked }
		federation = { remove_federation_flag = se_keyboard_locked }
		default = { log_error = "se_keyboard_lock_object: Cannot properly lock this scope type" }
	}
}
