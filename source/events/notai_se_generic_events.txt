 event = {
	id = se_setup.0
	is_triggered_only = yes
	hide_window = yes
	trigger = { NOT = { exists = event_target:se_global_event_database } }
	immediate = {
		create_species = {
			name = se_database_species_name
			class = MACHINE
			traits = {}
			effect = { se_save_as_global_event_target = { event_target_name = se_database_entry_species } }
		}
		se_create_database = yes
		event_target:se_last_created_database = {
			set_variable = {
				which = se_global_database_count
				value = 1
			}
			se_set_database_flag = { flag = se_global_database }
			se_save_as_global_event_target = { event_target_name = se_global_database }
		}
		se_create_event_country = { identifier = se_private_event_country }
		event_target:se_return_value = { se_save_as_global_event_target = { event_target_name = se_event_country } }
		se_clear_return_target = yes
	}
}

event = {
	id = se_setup.1
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		set_global_flag = script_extender_enabled						# Generic mod installed key
		fire_on_action = { on_action = on_game_start_begin }
		# Done for compatibility with vanilla
		if = {
			limit = { exists = event_target:global_event_country }
			clear_global_event_target = global_event_country
		}
		every_country = {
			limit = { is_country_type = global_event }
			destroy_country = yes
		}
	}
}

namespace = se_mod_auto_update

### On game loaded, register mod version
# if mod version doesn't already exist
# fire on_mod_self_update with that version number and the last number it had found
# else do nothing

# Initial register event
event = {
	id = se_mod_auto_update.0
	is_triggered_only = yes
	hide_window = yes
	trigger = { has_global_flag = se_mod_update_enabled }
	immediate = {
		fire_on_action = { on_action = on_se_register_mod_version }
	}
}

	id = se_rng.0
	is_triggered_only = yes
	hide_window = yes
	weight_multiplier = { factor = 1 }

		OR = {
			has_country_flag = se_rng_disable_min_checks

				value < 1
			}
		}
	}
	immediate = {
		set_variable = {
			which = se_rng_column_value
			value = 0
		}
	}
